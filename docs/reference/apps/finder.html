>>> import PyXA
>>> app = PyXA.application("Finder")
>>> window = app.windows()[0]
>>> lock = False
>>> (old_w, old_h) = (0,0)
>>> while True:
>>>     if window.position.y < 50 and lock is False:
>>>         # Increase height of window when user drags it to the top
